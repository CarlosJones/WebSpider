    这道题是数组做静态线性表的应用，外加一点排序。
    需要三个线性表：一个三维数组记录每个状态为C或I的条目，一维下标为队号，二维下标为题号，第三维两个元素，第一个表征是否已解决，若已解决存放第一次解决的时间，第二个记录在第一次解决前有几次Incorrect的提交；一个一维数组，记录每个队是否提交了；一个二维数组记录全部有提交记录的队伍的成绩，第二维有3个元素，第一个是队号，第二个是解了几道题，第三个是累计时间。
    处理输入，更新前两个线性表。根据这两个线性表，更新第三个，然后再将它按规则排序后，输出结果。
    使用合并排序，因为n的规模到了100，交换与比较（三元组）又较耗时，所以合并排序会比插入排序省时。
    代码如下： 
    #include <stdio.h>
#include <string.h>
static int comp(int a[3], int b[3])
{
	if (a[1] != b[1])
		return a[1] - b[1];
	else if (a[2] != b[2])
		return b[2] - a[2];
	else
		return b[0] - a[0];
}
static void merge(int (*A)[3], int p, int q, int r)
{
	int n1 = q - p + 1;
	int n2 = r - q;
	int L[n1 + 2][3];
	int R[n2 + 2][3];
	int i, j, k;

	for (i = 1; i <= n1; i++) {
		L[i][0] = A[p + i -1][0];
		L[i][1] = A[p + i -1][1];
		L[i][2] = A[p + i -1][2];
	}
	for (j = 1; j <= n2; j++) {
		R[j][0] = A[q + j][0];
		R[j][1] = A[q + j][1];
		R[j][2] = A[q + j][2];
	}
	L[i][0] = 101;
	L[i][1] = 0;
	L[i][2] = 0;
	R[j][0] = 101;
	R[j][1] = 0;
	R[j][2] = 0;
	i = 1;
	j = 1;
	for (k = p; k <= r; k++) {
		if (comp(L[i], R[j]) > 0) {
			A[k][0] = L[i][0];
			A[k][1] = L[i][1];
			A[k][2] = L[i][2];
			i++;
		} else {
			A[k][0] = R[j][0];
			A[k][1] = R[j][1];
			A[k][2] = R[j][2];
			j++;
		}
	}
}
static void merge_sort(int (*A)[3], int p, int r)
{
	if (r > p) {
		int q = (p + r) / 2;
		merge_sort(A, p, q);
		merge_sort(A, q + 1, r);
		merge(A, p, q, r);
	}
}

		for (i = 1; i <= 100; i++) {
			if (!sd[i])
				continue;
			nr_sd++;
			scores[nr_sd][0] = i;
			solved = 0;
			ptime = 0;
			for (j = 1; j <= 9; j++)
				if (actions[i][j][0] > 0) {
					solved++;
					ptime += actions[i][j][0];
					ptime += actions[i][j][1] * 20;
				}
			scores[nr_sd][1] = solved;
			scores[nr_sd][2] = ptime;
		}
		merge_sort(scores, 1, nr_sd);

    1个二维动态线性表和1个一维动态线性表的混合应用．N层（N>=4）遍历，且遍历过程中伴随着删除节点、添加节点的操作，即对线性表的改变。因此，在动态线性表的操作上,比第十二题解密要复杂（解密的动态线性表在遍历过程中无变化）．
    但是在解法上，虽然是多层遍历，比一般的深度要深，但还只是循环遍历，没有用到回溯，因此在解法上比十二题简单．总体比第十二题略容易．
    算法是：
    1. 按行放入paper动态线性表P,第i行P[i]下再挂一个存放该paper作者的动态线性表
    2. 初始化爱多士数作者动态线性表B,开头为爱多士自己, en←0
    3. 遍历B中爱多士数为en的B[j],对于每个包含B[j]的paper，将其中不在B中的作者放入B尾部，且将其爱多士数设为en+1,并对新增个数计数，遍历完该paper后删除该paper。
    4. 如果个数不为0，则en←en+1，回到第3步。
    5. 对于给出的要查询其爱多士数的每个作者，查找B，如果找到就打印其爱多士数，否则打印infinity
    对于第三步，遍历完该paper后删除该paper，再细说一下。第一点，因为处理完该paper后，它的作者要么被新添加进爱多士数线性表中，要么已经存在，不会遗漏一个，所以它没有存在的价值了，反而会变成拖累，如果不删除掉，则下次遍历的结果是所有的作者已经在爱多士线性表中了，做的是无用功，所以应当删除它(见代码152-155行)，提高下次查询的效率。第二点，在查找该作者是否已在爱多士数线性表中时，是否要从头开始查找呢？答案是不必的，只要从当前的B[j]开始查找就可以了(见代码143行)，这是因为之前的B中的作者不可能出现在该paper的作者中的，因为既然他们已经被处理过的，则包含他们的paper就被从P中删除掉了，即当前paper既然能出现，就说明不可能包含B[j]之前的作者，否则的话该paper早就被删除掉了而不可能出现。这样那个删除操作又带来了缩短查询爱多士数线性表的可能性（实际上，总运行时间又缩短了约1/4）。

    需要三个线性表：一个三维数组记录每个状态为C或I的条目，一维下标为队号，二维下标为题号，第三维两个元素，第一个表征是否已解决，若已解决存放第一次解决的时间，第二个记录在第一次解决前有几次Incorrect的提交；一个一维数组，记录每个队是否提交了；一个二维数组记录全部有提交记录的队伍的成绩，第二维有3个元素，第一个是队号，第二个是解了几道题，第三个是累计时间。

    使用合并排序，因为n的规模到了100，交换与比较（三元组）又较耗时，所以合并排序会比插入排序省时。

    #include <stdio.h>
#include <string.h>
static int comp(int a[3], int b[3])
{
	if (a[1] != b[1])
		return a[1] - b[1];
	else if (a[2] != b[2])
		return b[2] - a[2];
	else
		return b[0] - a[0];
}
static void merge(int (*A)[3], int p, int q, int r)
{
	int n1 = q - p + 1;
	int n2 = r - q;
	int L[n1 + 2][3];
	int R[n2 + 2][3];
	int i, j, k;

	for (i = 1; i <= n1; i++) {
		L[i][0] = A[p + i -1][0];
		L[i][1] = A[p + i -1][1];
		L[i][2] = A[p + i -1][2];
	}
	for (j = 1; j <= n2; j++) {
		R[j][0] = A[q + j][0];
		R[j][1] = A[q + j][1];
		R[j][2] = A[q + j][2];
	}
	L[i][0] = 101;
	L[i][1] = 0;
	L[i][2] = 0;
	R[j][0] = 101;
	R[j][1] = 0;
	R[j][2] = 0;
	i = 1;
	j = 1;
	for (k = p; k <= r; k++) {
		if (comp(L[i], R[j]) > 0) {
			A[k][0] = L[i][0];
			A[k][1] = L[i][1];
			A[k][2] = L[i][2];
			i++;
		} else {
			A[k][0] = R[j][0];
			A[k][1] = R[j][1];
			A[k][2] = R[j][2];
			j++;
		}
	}
}
static void merge_sort(int (*A)[3], int p, int r)
{
	if (r > p) {
		int q = (p + r) / 2;
		merge_sort(A, p, q);
		merge_sort(A, q + 1, r);
		merge(A, p, q, r);
	}
}

		memset(sd, 0, sizeof(sd));
		memset(actions, 0, sizeof(actions));

			ungetc(c, stdin);
			scanf("%d %d %d ", &c, &p, &t);
			sd[c] = 1;
			l = getchar();
			if (l == 'C' && !actions[c][p][0])
				actions[c][p][0] = t;
			else if (l == 'I' && !actions[c][p][0])
				actions[c][p][1] += 1;
			if (getchar() == EOF)
				break;
		}
		nr_sd = 0;
		for (i = 1; i <= 100; i++) {
			if (!sd[i])
				continue;
			nr_sd++;
			scores[nr_sd][0] = i;
			solved = 0;
			ptime = 0;
			for (j = 1; j <= 9; j++)
				if (actions[i][j][0] > 0) {
					solved++;
					ptime += actions[i][j][0];
					ptime += actions[i][j][1] * 20;
				}
			scores[nr_sd][1] = solved;
			scores[nr_sd][2] = ptime;
		}
		merge_sort(scores, 1, nr_sd);
		for (i = 1; i <= nr_sd; i++)
			printf("%d %d %d\n", scores[i][0], scores[i][1], scores[i][2]);
		if (cases > 0)
			putchar('\n');
	}

}

    1个二维动态线性表和1个一维动态线性表的混合应用．N层（N>=4）遍历，且遍历过程中伴随着删除节点、添加节点的操作，即对线性表的改变。因此，在动态线性表的操作上,比第十二题解密要复杂（解密的动态线性表在遍历过程中无变化）．
    但是在解法上，虽然是多层遍历，比一般的深度要深，但还只是循环遍历，没有用到回溯，因此在解法上比十二题简单．总体比第十二题略容易．
    算法是：
    1. 按行放入paper动态线性表P,第i行P[i]下再挂一个存放该paper作者的动态线性表
    2. 初始化爱多士数作者动态线性表B,开头为爱多士自己, en←0
    3. 遍历B中爱多士数为en的B[j],对于每个包含B[j]的paper，将其中不在B中的作者放入B尾部，且将其爱多士数设为en+1,并对新增个数计数，遍历完该paper后删除该paper。
    4. 如果个数不为0，则en←en+1，回到第3步。
    5. 对于给出的要查询其爱多士数的每个作者，查找B，如果找到就打印其爱多士数，否则打印infinity
    对于第三步，遍历完该paper后删除该paper，再细说一下。第一点，因为处理完该paper后，它的作者要么被新添加进爱多士数线性表中，要么已经存在，不会遗漏一个，所以它没有存在的价值了，反而会变成拖累，如果不删除掉，则下次遍历的结果是所有的作者已经在爱多士线性表中了，做的是无用功，所以应当删除它(见代码152-155行)，提高下次查询的效率。第二点，在查找该作者是否已在爱多士数线性表中时，是否要从头开始查找呢？答案是不必的，只要从当前的B[j]开始查找就可以了(见代码143行)，这是因为之前的B中的作者不可能出现在该paper的作者中的，因为既然他们已经被处理过的，则包含他们的paper就被从P中删除掉了，即当前paper既然能出现，就说明不可能包含B[j]之前的作者，否则的话该paper早就被删除掉了而不可能出现。这样那个删除操作又带来了缩短查询爱多士数线性表的可能性（实际上，总运行时间又缩短了约1/4）。

static struct en_author *find(const char *name, struct en_author *head)
{
       	struct en_author *enau;

	for (enau = head; enau != NULL; enau = enau->next)
	       	if (strcmp(name, enau->name) == 0)
		       	return enau;
       	return NULL;
}
static void free_papers(struct paper *head)
{
	struct paper *node;
	struct author *au;

	while (head->np != NULL) {
		node = head->np;
		head->np = node->np;
		while (node->na != NULL) {
			au = node->na;
			node->na = au->na;
			free(au->name);
			free(au);
		}
		free(node);
	}
	free(head);
}
static void free_paper(struct paper *p)
{
	struct author *au;

	while (p->na != NULL) {
		au = p->na;
		p->na = au->na;
		free(au);
	}
	free(p);
}
static void free_ens(struct en_author *head)
{
	struct en_author *enau;

	while (head->next != NULL) {
		enau = head->next;
		head->next = enau->next;
		free(enau->name);
		free(enau);
	}
	free(head);
}

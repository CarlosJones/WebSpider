    练习4-1: echo服务器
    这里使用spawn/3生成子进程，需要将loop/0导出，否则spawn调用不到loop/0 
    -module(echo).
-export([start/0, print/1, stop/0, loop/0]).

start() ->
    register(echo, spawn(echo, loop, [])).

print(Msg) ->
    echo ! {print, Msg}.

stop() ->
    echo ! stop.

loop() ->
    receive
        {print, Msg} ->
            io:format("~w~n", [Msg]),
            loop();
        stop ->
            true
    end.
    如果使用spawn/1，就不需要将loop/0导出。相关的改动如下:
    -export([start/0, print/1, stop/0]).
register(echo, spawn(fun() -> loop() end)).
